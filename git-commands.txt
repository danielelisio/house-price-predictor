# Option 2: Unix-style tree equivalent (instructor style)
function Show-Tree {
    param(
        [string]$Path = ".",
        [string[]]$Exclude = @(".venv", "__pycache__", ".git", "node_modules", ".ipynb_checkpoints")
    )
    
    function Show-Directory($dir, $prefix = "", $isLast = $true) {
        $items = Get-ChildItem $dir | Where-Object { 
            $exclude -notcontains $_.Name 
        } | Sort-Object Name
        
        for ($i = 0; $i -lt $items.Count; $i++) {
            $item = $items[$i]
            $isLastItem = ($i -eq ($items.Count - 1))
            $connector = if ($isLastItem) { "â””â”€â”€ " } else { "â”œâ”€â”€ " }
            
            Write-Host "$prefix$connector$($item.Name)" -ForegroundColor $(if ($item.PSIsContainer) { "Cyan" } else { "White" })
            
            if ($item.PSIsContainer) {
                $newPrefix = $prefix + $(if ($isLastItem) { "    " } else { "â”‚   " })
                Show-Directory $item.FullName $newPrefix $isLastItem
            }
        }
    }
    
    Write-Host (Split-Path $Path -Leaf) -ForegroundColor Yellow
    Show-Directory $Path
}
